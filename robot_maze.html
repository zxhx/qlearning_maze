import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):    
    action = None
    rand = random.randrange(0, 100, 1)
    if rand < epsilon*100: # 
        action = actions[random.randrange(0,4,1)] # 
    else:              #  Q 
        max_q = 0 
        for i in qline:
            if max_q < qline[i]:
                max_q = qline[i]
                action = i
        
    return action
